<Headings.h4 className="text-white bg-black inline-block px-4 py-2 rounded-full">validateRadio</Headings.h4>

<Paragraph className="mt-2 text-gray-400">
The `validateRadio` function validates a specified radio input field. It checks if a radio button within a group is selected and returns an <Link href="/docs/errorMessageInterface" className="text-blue-600 font-semibold" >ErrorMessageInterface</Link>.
</Paragraph>

<WorksIn visibleTypes={['html5', 'reactJS']} />

<ParameterList
  parameters={[
    {
      name: 'radioInputField',
      description: 'The radio input field or its ID to validate.',
      type: 'HTMLInputElement | string',
      dataType: 'HTMLElement | string',
      defaultValue: 'undefined',
      returnType: 'boolean | ErrorMessageInterface'
    },
    {
      name: 'customErrorMessage',
      description: 'Custom error message to return if validation fails.',
      type: 'object',
      dataType: 'object',
      defaultValue: '{}',
      returnType: 'boolean | ErrorMessageInterface'
    }
  ]}
/>

<Headings.h5 className="mt-4">Usage</Headings.h5>

<Headings.h6 className="mt-4">Example: Validate a Radio Button Group</Headings.h6>

<CodeBlock>
{`// Using with a radio input element
const radioElement = document.querySelector('input[name="gender"]');
const result = nfsfu234FormValidation.validateRadio(radioElement, {
  message: 'Please select a gender.'
});
console.log(result);

// Using with a radio input element ID
const resultById = nfsfu234FormValidation.validateRadio('genderRadioId', {
  message: 'Please select a gender.'
});
console.log(resultById);
`}
</CodeBlock>

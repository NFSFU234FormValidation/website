<Headings.h4 className="text-white bg-black inline-block px-4 py-2 rounded-full">displayError</Headings.h4>

<Paragraph className="mt-2 text-gray-400">
The `displayError` function is used to display error messages in either an `inline` or `modal` format. It allows specifying the duration for which the error message should be visible and associates the error message with a specific HTML element.
</Paragraph>

<WorksIn visibleTypes={['html5', 'reactJS']} />

<ParameterList
  parameters={[
    {
      name: 'details',
      description: 'The details of the error message to be displayed.',
      type: 'Object',
      dataType: 'object',
      defaultValue: 'undefined'
    },
    {
      name: 'details.type',
      description: 'The type of error display (either "inline" or "modal").',
      type: 'string',
      dataType: 'string',
      defaultValue: 'undefined'
    },
    {
      name: 'details.message',
      description: 'The error message to be displayed.',
      type: 'string',
      dataType: 'string',
      defaultValue: 'undefined'
    },
    {
      name: 'details.duration',
      description: 'The duration (in milliseconds) for which the error message should be visible.',
      type: 'number',
      dataType: 'number',
      defaultValue: 'undefined'
    },
    {
      name: 'details.element',
      description: 'The HTML element to which the error message is associated.',
      type: 'HTMLInputElement | HTMLSelectElement | HTMLTextAreaElement',
      dataType: 'HTMLElement',
      defaultValue: 'undefined'
    },
    {
      name: 'details.success',
      description: 'A flag indicating whether the operation was successful. Default is false.',
      type: 'boolean',
      dataType: 'boolean',
      defaultValue: 'false'
    }
  ]}
/>

<CodeBlock>

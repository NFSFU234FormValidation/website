<Headings.h4 className="text-white bg-black inline-block px-4 py-2 rounded-full">AJAX</Headings.h4>

<Paragraph className="mt-2 text-gray-400">
The `ajax` function is used to make asynchronous HTTP requests. This function provides a structured way to handle AJAX requests with customizable options for headers, request methods, and request bodies.
</Paragraph>

<WorksIn visibleTypes={['html5', 'reactJS', 'nodejs']} />

<ParameterList
  parameters={[
    {
      name: 'AJAXOptions',
      description: 'Options for the AJAX request. This includes URL, request method, headers, request body, and other configurations.',
      type: '{ url?: string; RequestMethod?: string; RequestHeader?: { [key: string]: string }; RequestBody?: any; RequestBodyIgnore?: boolean; RequestType?: string }',
      dataType: 'object',
      defaultValue: 'undefined',
      returnType: 'Promise<any>'
    },
  ]}
/>

<Headings.h5 className="mt-4">Usage</Headings.h5>

<Headings.h6 className="mt-4">Example 1: GET Request</Headings.h6>

<CodeBlock>

{`const AJAXOptions = {
    url: 'https://your-api-endpoint.tld', // URL for the AJAX request
    RequestMethod: 'GET', // Request method
    RequestHeader: {
        'Content-Type': 'application/json', // Example request header
        // Add other request headers if needed
    },
    RequestBodyIgnore: true
};

nfsfu234FormValidation.ajax(AJAXOptions)
    .then((response) => {
        // Success: Server response received in JSON format
        console.log('Request successful', response);
    })
    .catch((error) => {
        // Error: AJAX request failed or rejected
        console.error('Request failed', error);
    });
`}
</CodeBlock>

<Headings.h6 className="mt-4">Example 2: POST Request</Headings.h6>

<CodeBlock>
{`const AJAXOptions = {
    url: 'https://your-api-endpoint.tld', // URL for the AJAX request
    RequestMethod: 'POST', // Request method
    RequestHeader: {
        'Content-Type': 'application/json', // Example request header
        // Add other request headers if needed
    },
    RequestBody: {
        title: 'new product',
        price: 29.99,
        description: 'A great product',
        image: 'https://i.pravatar.cc',
        category: 'electronics'

nfsfu234FormValidation.ajax(AJAXOptions)
    .then((response) => {
        // Success: Server response received in JSON format
        console.log('Request successful', response);
    })
    .catch((error) => {
        // Error: AJAX request failed or rejected
        console.error('Request failed', error);
    });
`}
</CodeBlock>

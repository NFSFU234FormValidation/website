<Headings.h4 className="text-white bg-black inline-block px-4 py-2 rounded-full">generatePassword</Headings.h4>

<Paragraph className="mt-2 text-gray-400">
The `generatePassword` function generates a random password of specified length, optionally hashes the password, and returns the plain and hashed passwords if hashing is enabled.
</Paragraph>

<WorksIn visibleTypes={['html5', 'reactJS', 'nodejs']} />

<ParameterList
  parameters={[
    {
      name: 'length',
      description: 'The length of the generated password. If not provided, a random length between 8 and 18 characters will be used.',
      type: 'number',
      dataType: 'number',
      defaultValue: 'Random value between 8 and 18',
      returnType: 'Promise<string | {password: string, hashedPassword: string}>'
    },
    {
      name: 'shouldHash',
      description: 'A flag indicating whether the generated password should be hashed. Default is false.',
      type: 'boolean',
      dataType: 'boolean',
      defaultValue: 'false',
      returnType: 'Promise<string | {password: string, hashedPassword: string}>'
    }
  ]}
/>

<Headings.h5 className="mt-4">Usage</Headings.h5>

<Headings.h6 className="mt-4">Example: Generate Random Password</Headings.h6>

<CodeBlock>
{`const randomPassword = nfsfu234FormValidation.generatePassword();
randomPassword.then(password => {
    console.log("Generated Password: ", password); // Example output: "A!3d7fhJ"
});
`}
</CodeBlock>

<Headings.h6 className="mt-4">Example: Generate Random Password with Hashing</Headings.h6>

<CodeBlock>
{`const randomPasswordWithHash = nfsfu234FormValidation.generatePassword(12, true);
randomPasswordWithHash.then(passwordInfo => {
    const { password, hashedPassword } = passwordInfo;
    console.log("Generated Password: ", password); // Example output: "B@9k2zLwQp4m"
    console.log("Hashed Password: ", hashedPassword); // Example output: "$2a$10$eBzFgqRvKjZtqJjvZvQXXO..."
});
`}
</CodeBlock>

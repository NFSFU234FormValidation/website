<Headings.h4 className="text-white bg-black inline-block px-4 py-2 rounded-full">validateTextarea</Headings.h4>

<Paragraph className="mt-2 text-gray-400">
The `validateTextarea` function validates a specified textarea input field. It checks if the textarea meets the validation criteria defined in the `options` and returns an <Link href="/docs/errorMessageInterface" className="text-blue-600 font-semibold" >ErrorMessageInterface</Link>.
</Paragraph>

<WorksIn visibleTypes={['html5', 'reactJS']} />

<ParameterList
  parameters={[
    {
      name: 'textareaField',
      description: 'The textarea input field to validate.',
      type: 'HTMLTextAreaElement',
      dataType: 'HTMLElement',
      defaultValue: 'undefined',
      returnType: 'boolean | ErrorMessageInterface'
    },
    {
      name: 'options',
      description: 'Optional settings for validation such as custom error messages.',
      type: 'object',
      dataType: 'object',
      defaultValue: '{}',
      returnType: 'boolean | ErrorMessageInterface'
    }
  ]}
/>

<Headings.h5 className="mt-4">Usage</Headings.h5>

<Headings.h6 className="mt-4">Example: Validate a Textarea Input</Headings.h6>

<CodeBlock>
{`// Using with a textarea input element
const textareaElement = document.querySelector('textarea');
const result = nfsfu234FormValidation.validateTextarea(textareaElement, {
  message: 'Please provide a valid input.'
});
console.log(result);
`}
</CodeBlock>

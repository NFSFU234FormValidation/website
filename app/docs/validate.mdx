<Headings.h4 className="text-white bg-black inline-block px-4 py-2 rounded-full">Validate</Headings.h4>

<Paragraph className="mt-2 text-gray-400">
The `validate` function is used to validate a form. When you call this function, it will automatically validate the form based on the specified rules and return the validation result.
</Paragraph>

<WorksIn visibleTypes={['html5', 'reactJS']} />

  parameters={[

      returnType: 'boolean | Promise<any>'
    },
    {

      returnType: 'boolean | Promise<any>'
    },
  ]}
/>

<Headings.h5 className="mt-4">Usage</Headings.h5>

<Headings.h6 className="mt-4">Option 1.</Headings.h6>

<CodeBlock>

{`const myForm = document.getElementById("myForm")
nfsfu234FormValidation.validate(form)
.then((response) => {
    const { message, type, data } = response;
    console.log(response);
});`}

</CodeBlock>

<Headings.h6 className="mt-4">Option 2.</Headings.h6>

<CodeBlock>
{`const myForm = document.getElementById("myForm");
nfsfu234FormValidation.validate(myForm, (response) => {
    const { message, type, data } = response;
    console.log(response);
});`}
</CodeBlock>

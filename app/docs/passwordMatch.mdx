<Headings.h4 className="text-white bg-black inline-block px-4 py-2 rounded-full">passwordMatch</Headings.h4>

<Paragraph className="mt-2 text-gray-400">
The `passwordMatch` function checks if a plain text password matches a hashed password. It can use bcrypt for hashed comparisons or directly compare plain text passwords.
</Paragraph>

<WorksIn visibleTypes={['nodejs', 'reactJS', 'html5']} />

<ParameterList
  parameters={[
    {
      name: 'stringPassword',
      description: 'The plain text password to verify.',
      type: 'string',
      dataType: 'string',
      defaultValue: 'undefined',
      returnType: 'Promise<boolean>'
    },
    {
      name: 'hashedPassword',
      description: 'The hashed password to compare against.',
      type: 'string',
      dataType: 'string',
      defaultValue: 'undefined',
      returnType: 'Promise<boolean>'
    },
    {
      name: 'isHashed',
      description: 'Indicates whether the provided password is already hashed. Default is false.',
      type: 'boolean',
      dataType: 'boolean',
      defaultValue: 'false',
      returnType: 'Promise<boolean>'
    }
  ]}
/>

<Headings.h5 className="mt-4">Usage</Headings.h5>

<Headings.h6 className="mt-4">Example: Verify Plain Text Password</Headings.h6>

<CodeBlock>
{`(async () => {
  try {
    const stringPassword = 'Password123';
    const hashedPassword = '@Password123';
    const isMatch = await nfsfu234FormValidation.passwordMatch(stringPassword, hashedPassword);
    console.log("Passwords Match: ", isMatch); // true or false
  } catch (error) {
    console.error("Error verifying password: ", error);
  }
})();
`}
</CodeBlock>

<Headings.h6 className="mt-4">Example: Verify Hashed Password</Headings.h6>

<CodeBlock>
{`(async () => {
  try {
    const stringPassword = 'Password123';
    const hashedPassword = '$2a$10$SJVY.s8OD3S1D3HJh3n3MeI8yzUmJm0bO5L2R3Z78b8MWg8L4NLqG';
    const isMatch = await nfsfu234FormValidation.passwordMatch(stringPassword, hashedPassword, true);
    console.log("Passwords Match: ", isMatch); // true or false
  } catch (error) {
    console.error("Error verifying password: ", error);
  }
})();
`}
</CodeBlock>

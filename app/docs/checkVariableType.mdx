<Headings.h4 className="text-white bg-black inline-block px-4 py-2 rounded-full">checkVariableType</Headings.h4>

<Paragraph className="mt-2 text-gray-400">
The `checkVariableType` function checks the type of a given variable and returns a string that represents its type. If the type cannot be determined, it returns `'unknown'`.
</Paragraph>

<WorksIn visibleTypes={['html5', 'reactJS', 'nodejs']} />

<ParameterList
  parameters={[
    {
      name: 'variable',
      description: 'The variable whose type needs to be checked.',
      type: 'any',
      dataType: 'any',
      defaultValue: 'undefined',
      returnType: 'string'
    }
  ]}
/>

<Headings.h5 className="mt-4">Usage</Headings.h5>

<Headings.h6 className="mt-4">Example: Check Variable Type</Headings.h6>

<CodeBlock>
{`const type1 = nfsfu234FormValidation.checkVariableType("Hello"); // "string"
const type2 = nfsfu234FormValidation.checkVariableType(123);    // "number"
const type3 = nfsfu234FormValidation.checkVariableType([]);     // "array"
const type4 = nfsfu234FormValidation.checkVariableType({});     // "object"
const type5 = nfsfu234FormValidation.checkVariableType(() => {}); // "function"
const type6 = nfsfu234FormValidation.checkVariableType(null);   // "null"
const type7 = nfsfu234FormValidation.checkVariableType(undefined); // "undefined"
const type8 = nfsfu234FormValidation.checkVariableType(document.body); // "HTML Element"`}
</CodeBlock>

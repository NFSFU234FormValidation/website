<Headings.h4 className="text-white bg-black inline-block px-4 py-2 rounded-full ">validateAllInput</Headings.h4>

<Paragraph className="mt-2 text-gray-400">
The `validateAllInput` function validates all input fields within a specified form and returns an array of error messages (a nested <Link href="/docs/errorMessageInterface" className="text-blue-600 font-semibold" >ErrorMessageInterface</Link>) if invalid, or a single instance of <Link href="/docs/errorMessageInterface" className="text-blue-600 font-semibold" >ErrorMessageInterface</Link> if all fields are valid.
</Paragraph>

<WorksIn visibleTypes={['html5', 'reactJS']} />

<ParameterList
  parameters={[
    {
      name: 'form',
      description: 'The form element or its ID.',
      type: 'HTMLFormElement | HTMLDivElement | string',
      dataType: 'HTMLElement | string',
      defaultValue: 'undefined',
      returnType: 'ErrorMessageInterface[] | boolean'
    },
    {
      name: 'options',
      description: 'Additional options for validation.',
      type: 'object',
      dataType: 'object',
      defaultValue: '{}',
      returnType: 'ErrorMessageInterface[] | boolean',
      properties: [
        {
          name: 'customErrorMessages',
          description: 'Custom error messages for form validation.',
          type: 'object',
          defaultValue: '[]',
        },
        {
          name: 'form',
          description: 'The form element associated with the input fields.',
          type: 'HTMLElement',
          defaultValue: 'undefined',
        }
      ]
    }
  ]}
/>

<Headings.h5 className="mt-4">Usage</Headings.h5>

<Headings.h6 className="mt-4">Example: Validate All Input Fields in a Form</Headings.h6>

<CodeBlock>
{`const formElement = document.getElementById('myForm');

nfsfu234FormValidation.validateAllInput(formElement, {
  customErrorMessages: {
    email: 'Please enter a valid email address.',
    url: 'Please enter a valid URL.',
    date: 'Please enter a valid date.',
  }
}).then(response => {
  console.log(response);
});
`}
</CodeBlock>

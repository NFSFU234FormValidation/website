<Headings.h4 className="text-white bg-black inline-block px-4 py-2 rounded-full">validateInput</Headings.h4>

<Paragraph className="mt-2 text-gray-400">
The `validateInput` function validates a specified input field and returns a promise that resolves to the <Link href="/docs/errorMessageInterface" className="text-blue-600 font-semibold" >ErrorMessageInterface</Link>.
</Paragraph>

<Headings.h5 className="mt-4">Usage</Headings.h5>

<Headings.h6 className="mt-4">Example: Validate an Email Input Field</Headings.h6>

<CodeBlock>
{`// Using with an input element
const inputElement = document.querySelector('input[type="email"]');
nfsfu234FormValidation.validateInput(inputElement, {
  errorType: 'inline',
  customErrorMessages: {
    email: {
        empty : 'Please enter an email address.',
        formart : 'Please enter a valid email address.'
    }
  }
}).then(response => {
  console.log(response);
});

// Using with an input element ID
nfsfu234FormValidation.validateInput('emailInputId', {
  errorType: 'modal',
  customErrorMessages: {
    email: {
        empty : 'Please enter an email address.',
        formart : 'Please enter a valid email address.'
    }
  }
}).then(response => {
  console.log(response);
});
`}
</CodeBlock>

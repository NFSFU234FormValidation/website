<Headings.h4 className="text-white bg-black inline-block px-4 py-2 rounded-full">ErrorMessageInterface</Headings.h4>

<Paragraph className="mt-2 text-gray-400">
The `ErrorMessageInterface` defines the structure of the error messages returned by the `validation`, `ajax`, etc. functions in the NFSFU234FormValidation Library.
</Paragraph>

<ParameterList
  parameters={[
    {
      name: 'message',
      description: 'The error message which can be a string, boolean, or number.',
      type: 'string | boolean | number',
      defaultValue: 'undefined'
    },
    {
      name: 'type',
      description: 'The type of error.',
      type: 'string',
      defaultValue: 'undefined'
    },
    {
      name: 'code',
      description: 'The error code.',
      type: 'number',
      defaultValue: 'undefined'
    },
    {
      name: 'data',
      description: 'Additional data related to the error which can be of various types.',
      type: 'string | number | object | Array<any> | JSON | HTMLElement | null | undefined | boolean',
      defaultValue: 'undefined'
    }
  ]}
/>

<Headings.h5 className="mt-4">Usage</Headings.h5>

<Headings.h6 className="mt-4">Example: Handling an Error Message using the <Link href="/docs/validate" className="font-semibold text-blue-500" >`Validate API`</Link> function </Headings.h6>

<CodeBlock>
{`const myForm = document.getElementById("myForm");
nfsfu234FormValidation.validate(myForm, (response) => {
    /**
     *  Example of using the ErrorMessageInterface
     * 
     * Here the response is in the ErrorMessageInterface formart so we destructure the ErrorMessageInterface 
     * This enables us to to access its attributes and use them easily
    */
    
    const { message, type, data, code } = response;
    
    console.log("Code: ", code);
    console.log("Message: ", message);
    console.log("Type: ", type);
    console.log("Data: ", data);
});
`}
</CodeBlock>

<Headings.h5 className="mt-4">Usage</Headings.h5>

<Headings.h6 className="mt-4">Example: Setting Custom Error Messages using the <Link href="/docs/validate" className="font-semibold text-blue-500">`Validate API`</Link> function</Headings.h6>

<CodeBlock>
{`const customMessages = {
  email: {
    empty: "Please enter your email address.",
    invalid: "The email address you entered is not valid."
  },
  password: "Your password must be at least 8 characters long."
};

nfsfu234FormValidation.validate(myForm, { customErrorMessages: customMessages }, (response) => {
  if (!response.isValid) {
    console.log(response.errorMessages);
  }
});
`}
</CodeBlock>

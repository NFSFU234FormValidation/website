</Paragraph>

<Headings.h6 className="mt-4">Example: Validate Password</Headings.h6>

<CodeBlock>

`}
</CodeBlock>

<Headings.h6 className="mt-4">Example: Validate Password with Symbol Check</Headings.h6>

<CodeBlock>
{`const isStrongPassword = nfsfu234FormValidation.checkPassword('StrongP@ssword123', true);
console.log("isStrongPassword: ", isStrongPassword); // true or error message
`}
</CodeBlock>

<Headings.h6 className="mt-4">Example: Validate Password with Symbol Check With Custom Regex</Headings.h6>

<CodeBlock>

`}
</CodeBlock>

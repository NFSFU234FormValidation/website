<Headings.h4 className="text-white bg-black inline-block px-4 py-2 rounded-full" >Submit</Headings.h4>

<Paragraph className="mt-2 text-gray-400">
The `submit` function is used to submit a form. When you call this function, you are going to be automatically validating the form and then it will process for submitting. 
</Paragraph>

<WorksIn visibleTypes={['html5', 'reactJS',]} />

    },

    },
  ]}
/>

<Headings.h5 className="mt-4">Usage</Headings.h5>

<Headings.h6 className="mt-4">Option 1.</Headings.h6>

<CodeBlock>

{`const myForm = document.getElementById("myForm");
nfsfu234FormValidation.submit(myForm)
.then((response) => {
    const { message, type, data } = response
    console.log(response);
});`}

</CodeBlock>
